Posted on May 31, 2022
AUC
# Parameter Type Default Interpretation
1 y_true torch.tensor((n, 1)) None The true label of image.
2 y_pred torch.tensor((n, 1)) None The predict value of image.
3 multi_type String "ovo" Either "ova" and "ovo".
4 acc Bool True Whether to enable accelerated computing.
# Return Type
1 Score Float 
from XCurve.AUROC.metrics import AUC
import torch

target = torch.randint(0, 2, (64, 1)).numpy()
pred = torch.randn((64, 1)).numpy()

res = AUC(target, pred)
p2AUC
# Parameter Type Default Interpretation
1 y_true torch.tensor((n, 1)) None The true label of image.
2 y_pred torch.tensor((n, 1)) None The predict value of image.
3 min_tpr Float None The min value of true positive rate.
4 min_fpr Float None The max value of false positive rate.
# Return Type
1 Score Float 
from XCurve.AUROC.metrics import p2AUC
import torch

target = torch.randint(0, 2, (64, 1)).numpy()
pred = torch.randn((64, 1)).numpy()

res = p2AUC(target, pred, 0.3, 0.5)
